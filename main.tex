\begin{algorithm}[tb]
    \caption{配列から最小値を求める関数}
    \label{alg:min-from-array}
    \begin{algorithmic}[1]
        \Function{min\_in\_array}{$array$}
        \State $min \gets 0$
        \ForAll {$element \gets array$}
        \If {$element < min$}
        \State $min \gets element$
        \EndIf
        \EndFor
        \State \Return $min$
        \EndFunction
    \end{algorithmic}
\end{algorithm}
